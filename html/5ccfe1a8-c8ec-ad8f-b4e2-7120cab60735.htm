Remarks
Be careful to the underlying values. You had better set levels and labels argument explicitly.
Examples
Copy
public enum Group
{
   Treatment = 1,
   Control = 2
}

// You must set 'levels' and 'labels' arguments explicitly in this case
// because levels of factor is sorted by default and the names in R and in enum names are different.
var code = @"factor(
   c(rep('T', 5), rep('C', 5), rep('T', 4), rep('C', 5)),
   levels=c('T', 'C'),
   labels=c('Treatment', 'Control')
)";
var factor = engine.Evaluate(code).AsFactor();
foreach (Group g in factor.GetFactors<Group>())
{
   Console.Write("{0} ", g);
}
